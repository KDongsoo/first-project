function shuffle(array) { for (let i = array.length - 1; i > 0; i--) { const j = Math.floor(Math.random() * (i + 1)); [array[i], array[j]] = [array[j], array[i]]; } return array; } function generateNumbers(start, end) { const numbers = []; for (let i = start; i <= end; i++) { numbers.push(i); } return numbers; } function generateShuffledNumbers(start, end, count) { const numbers=generateNumbers(start, end); const shuffledNumbers=[]; while (shuffledNumbers.length < count) { const remaining=count - shuffledNumbers.length; const portion=Math.min(remaining, numbers.length); const portionNumbers=numbers.splice(0, portion); shuffledNumbers.push(...shuffle(portionNumbers)); } return shuffledNumbers; } const shuffledNumbers=generateShuffledNumbers(8, 16, 1000); console.log(shuffledNumbers);
